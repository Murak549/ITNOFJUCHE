			if = {
				limit = {
					country_exists = GER

						country_exists = UKR
					}

					add_ideas = cold_war
					add_ai_strategy = {
						type = support
						id = PREV
						value = 200
					}
				}
				PRC = {
					every_owned_state = {
						create_unit = {
							division = "division_template = \"Juntuán\" start_experience_factor = 0.5"
							owner = PRC
						}
						create_unit = {
							division = "division_template = \"Renmin Jundui\" start_experience_factor = 0.5"
							owner = PRC
						}
					}
				}
			}
            every_country = {
				limit = {
					NOT = {
						OR = {
							original_tag = PRK
							original_tag = ETH
							original_tag = MAN
							original_tag = JAP
							original_tag = GER
							original_tag = ENG
							original_tag = FRA
							original_tag = USA
							original_tag = SPR
							original_tag = POR
							original_tag = SWI
							original_tag = ITA
							original_tag = SOV

							original_tag = LAT
							original_tag = LIT

							original_tag = PRC
							original_tag = SIK
							original_tag = XSM

							original_tag = YUN
							original_tag = MEX
							original_tag = CAN
							original_tag = NZL
							original_tag = AST
							original_tag = SAF
						}
					}
				}
                load_focus_tree = generic_focus
            }
		}
        if = { #조선은 하나다
			limit = {
				OR = {
					AND = {
						original_tag = KOR
						has_war_with = PRK

					AND = {
						original_tag = HMC
						has_war_with = PRK
					}
				}
			}

				country_event = { id = prk.1 hours = 2 }
			}
		}
	}

		effect = {
			if = {
				limit = {
					tag = PRK
					FROM = { tag = KOR }
				}
				play_song = "songofpeoplearmy"
			}
			if = {
				limit = {
					tag = PRK
					FROM = { tag = JAP }
				}
				play_song = "sacred__war"
			}
			if = {
				limit = {
					tag = PRK
					FROM = { tag = PRC }
				}
				play_song = "strike_da"
				set_global_flag = PRK_true_sacred_war
			}
		}
	}
}
